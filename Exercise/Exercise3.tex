\begin{exercise}\label{ex3.15}
	Let $\left\{x_i\right\}_{i=1}^n$ be generators and $\left\{e_i\right\}_{i=1}^n$ be the canonical basis. Let $\phi(e_i)=x_i$, so $\phi$ is surjective. Consider a residue field $k=A/\mathfrak{m}$, we have $\operatorname{Tor}_1(k,F)\rightarrow k\otimes\ker\phi\rightarrow k\otimes F\xrightarrow{1\otimes\phi}k\otimes F\rightarrow0$. $\operatorname{Tor}_1(k,F)=0$ by balancing Tor theorem; $1\otimes\phi$ bijective so $k\otimes\ker\phi=0$. By \Cref{ex2.3}, $\ker\phi=0$.
	
	If $\left\{x_i\right\}_{i=1}^m$ generate $F$ where $m<n$, pick some element $\left\{x_i\right\}_{i=m+1}^n$ so we also have $\phi(e_i)=x_i$. By injectiveness of $\phi$, $\left\{e_i\right\}_{i=1}^n$ are linearly dependent.
\end{exercise}

\begin{exercise}\label{ex3.16}
	\begin{enumerate}
		\item[(i) $\Rightarrow$ (ii)]$f^*\left(p^e\right)=p$.
		\item[(ii) $\Rightarrow$ (iii)] Use \Cref{1.17}.
		\item[(iii) $\Rightarrow$ (iv)] Every module contains a cyclic submodule, which is isomorphic to $A/\mathfrak{a}$. Then use $B\otimes_AA/\mathfrak{a}\cong B/\mathfrak{a}^e$ and $\mathfrak{a}^e\subseteq\mathfrak{m}^e$.
		\item[(iv) $\Rightarrow$ (v)] Let the mapping be $\phi$. Since $B$ flat, $0\rightarrow(\ker\phi)_B\rightarrow M_B\rightarrow\left(M_B\right)_B$ exact. By \Cref{ex2.13}, $M_B\rightarrow\left(M_B\right)_B$ injective, so $(\ker\phi)_B=0$; by assumption, $\ker\phi=0$.
		\item[(v) $\Rightarrow$ (i)] Take $M=A/\mathfrak{a}$. Then $M_B=B/\mathfrak{a}^e$.
	\end{enumerate}
\end{exercise}

\begin{exercise}\label{ex3.17}
	If $M\rightarrow N$ injective, we have a diagram
	\begin{equation*}
		\begin{tikzcd}
			B\otimes_AM\arrow[r,"\phi"]\arrow[d,"g\otimes\operatorname{id}_M"]&B\otimes_AN\arrow[d,"g\otimes\operatorname{id}_N"]\\
			C\otimes_AM\arrow[r,"\psi"]&C\otimes_AN
		\end{tikzcd}
	\end{equation*}
	$g\circ f$ flat, so $\psi$ injective. $g$ faithfully flat, by \hyperref[ex3.16]{\Cref*{ex3.16}(v)}, $B\otimes_AM\rightarrow C\otimes_B\left(B\otimes_AM\right)$ injective. By \Cref{2.15}, $C\otimes_B\left(B\otimes_AM\right)=C\otimes_AM$, so $g\otimes\operatorname{id}_M$ injective.
\end{exercise}
