\section{Completion}

\begin{exercise}\label{ex10.1}
	Some completions are calculated as follows
	\begin{itemize}
		\item $A/p^nA=\bigoplus_{k\geqslant0}\mathbb{Z}/p\mathbb{Z}$, where the transition maps are all identity.
		\item $A_n=\alpha^{-1}\left(p^nB\right)=\bigoplus_{k>n}\mathbb{Z}/p\mathbb{Z}$, so $A/A_n=\bigoplus_{k=0}^n\mathbb{Z}/p\mathbb{Z}$. The transition maps mean, if $\left(x^{(1)},x^{(2)},\cdots\right)\in\varprojlim A/A_n$, then $x^{(i)}_i=x^{(i+1)}_i=\cdots$ for all $i$.
		\item $B/p^nB=\left(\bigoplus_{k=0}^n\mathbb{Z}/p^k\mathbb{Z}\right)\oplus\left(\bigoplus_{k>n}\mathbb{Z}/p^n\mathbb{Z}\right)$. The transition maps are similar to $A/A_n$.
	\end{itemize}
	The sequence $0\rightarrow A\xrightarrow{\alpha}B\xrightarrow{p\cdot}pB\rightarrow0$ is exact. Since $p$-adic topology and induced topology on $pB$ coincide, by \Cref{10.3}, $\ker\left(\widehat{p\cdot}\right)=\varprojlim A/A_n$.
\end{exercise}

\begin{exercise}\label{ex10.3}
	Denote $N=\bigcap_{\mathfrak{m}\supseteq\mathfrak{a}}\ker\left(M\rightarrow M_\mathfrak{m}\right)$. Let $x\in\bigcap_{n=1}^\infty\mathfrak{a}^nM$. By \Cref{10.17}, we have $(1+a)x=0$. If $\mathfrak{m}\supseteq\mathfrak{a}$, $1+a\in A-\mathfrak{m}$, so $x\in N$. Since $\left(\ker\left(M\rightarrow M_\mathfrak{m}\right)\right)_\mathfrak{m}=0$, we have $N_\mathfrak{m}=0$ for all $\mathfrak{m}\supseteq\mathfrak{a}$. By \Cref{ex3.14}, $N=\mathfrak{a}N$, and therefore $N=\mathfrak{a}N=\mathfrak{a}^2N=\cdots$. Thus, $N=\bigcap_{n=1}^\infty\mathfrak{a}^nN\subseteq\bigcap_{n=1}^\infty\mathfrak{a}^nM$.
	
	We can see $N$ is also equal to $\bigcap_{\mathfrak{p}\supseteq\mathfrak{a}}\ker\left(M\rightarrow M_\mathfrak{p}\right)$. Therefore, $\operatorname{Supp}(M)\cap V(\mathfrak{a})=0$ iff $N=M$, iff $M=\bigcap_{n=1}^\infty\mathfrak{a}^nM$, iff $M=\mathfrak{a}M$. By \Cref{ex2.2}, \Cref{10.13} and \Cref{10.15.3}, $\hat{M}/\hat{\mathfrak{a}}\hat{M}=\hat{A}/\hat{\mathfrak{a}}\otimes_{\hat{A}}\hat{M}=A/\mathfrak{a}\otimes_A\left(M\otimes_A\hat{A}\right)=M/\mathfrak{a}M\otimes_A\hat{A}$. Since $\hat{A}\neq0$, by \Cref{ex2.3}, $M/\mathfrak{a}M=0$ iff $\hat{M}/\hat{\mathfrak{a}}\hat{M}=0$. By \Cref{10.15.4} and \hyperref[2.6]{Nakayama's lemma}, $\hat{M}=\hat{a}\hat{M}$ iff $\hat{M}=0$.
\end{exercise}

\begin{exercise}\label{ex10.5}
	We can deduce from \Cref{10.2} that taking $\mathfrak{a}$-adic completion on $0\rightarrow b^mM\rightarrow M\rightarrow M/b^mM\rightarrow0$ preserves exactness. Thus, $\left(M/b^mM\right)^\mathfrak{a}\cong M^\mathfrak{a}/b^mM^\mathfrak{a}$, and therefore
	\begin{equation*}
		\left(M^\mathfrak{a}\right)^\mathfrak{b}
		=\varprojlim_m\left(\varprojlim_n\left.\left(M/\mathfrak{a}^nM\right)\middle/\left(\mathfrak{b}^m\left(M/\mathfrak{a}^nM\right)\right)\right.\right)
		=\varprojlim_m\left(\varprojlim_nM/\left(\mathfrak{a}^n+\mathfrak{b}^m\right)M\right).
	\end{equation*}
	
	On the other hand, we have $(\mathfrak{a}+\mathfrak{b})^{2n}\subseteq\mathfrak{a}^n+\mathfrak{b}^n\subseteq(\mathfrak{a}+\mathfrak{b})^{n}$. If $x\in(\mathfrak{a}+\mathfrak{b})^{n}$, $x+\mathfrak{a}^n+\mathfrak{b}^n\subseteq(\mathfrak{a}+\mathfrak{b})^{n}$, which means each set in the basis $\left\{(\mathfrak{a}+\mathfrak{b})^{n}\right\}$ is open in the topology generated by the basis $\left\{\mathfrak{a}^n+\mathfrak{b}^n\right\}$, and vice versa. Thus, they generate a same topology, and therefore
	\begin{equation*}
		M^{\mathfrak{a}+\mathfrak{b}}=\varprojlim M/(\mathfrak{a}+\mathfrak{b})^nM=\varprojlim M/\left(\mathfrak{a}^n+\mathfrak{b}^n\right)M.
	\end{equation*}
	
	We construct two homomorphisms $\begin{tikzcd}
		\varprojlim_m\left(\varprojlim_nM/\left(\mathfrak{a}^n+\mathfrak{b}^m\right)M\right)\arrow[r, shift left, "\Phi"]&
		\varprojlim M/\left(\mathfrak{a}^n+\mathfrak{b}^n\right)M\arrow[l, shift left, "\Psi"]
	\end{tikzcd}$. When $k\geqslant\max(m,n)$, $\mathfrak{a}^k+\mathfrak{b}^k\subseteq\mathfrak{a}^n+\mathfrak{b}^m$, so there is a canonical quotient map $\psi_{nm}:M/\left(\mathfrak{a}^k+\mathfrak{b}^k\right)M\rightarrow M/\left(\mathfrak{a}^n+\mathfrak{b}^m\right)M$. Let $\Phi\left((x)_{nm}\right)=(x)_{kk}$ be the diagonal map, and let $\Psi\left((y)_k\right)=\psi_{nm}\left((y)_k\right)$.

\end{exercise}

\begin{exercise}\label{ex10.6}
	If $\mathfrak{a}\subseteq\mathfrak{J}$, let $x\notin\mathfrak{m}$. We can find an open neighborhood $x+\mathfrak{a}$ of $x$ such that $(x+\mathfrak{a})\cap\mathfrak{m}=\varnothing$.
	
	By \Cref{10.1}, the closure of $\mathfrak{m}$ is $\bigcap_{n\geqslant1}\left(\mathfrak{m}+\mathfrak{a}^n\right)$. If $\mathfrak{a}\nsubseteq\mathfrak{m}$, we have $\mathfrak{m}+\mathfrak{a}^n=A$, otherwise $\mathfrak{a}\subseteq\sqrt{\mathfrak{a}^n}\subseteq\mathfrak{m}$.
\end{exercise}

\begin{exercise}\label{ex10.7}
	Let $M$ be a finitely generated $A$-module. By \Cref{10.13}, $\hat{M}=M_{\hat{A}}$, so $\ker\left(M\rightarrow M_{\hat{A}}\right)=\bigcap_{n=1}^\infty\mathfrak{a}^nM$.
	
	If $\mathfrak{a}\subseteq\mathfrak{J}$, every element in $1+\mathfrak{a}$ is a unit, and by \Cref{10.17}, $\bigcap_{n=1}^\infty\mathfrak{a}^nM=0$. If $N$ is an arbitrary $A$-module and $x\in\ker\left(N\rightarrow N_{\hat{A}}\right)$, then $x\in\ker\left(Ax\rightarrow (Ax)_{\hat{A}}\right)$ and $Ax$ is finitely generated.
	
	If $\mathfrak{a}\nsubseteq\mathfrak{m}$, we have $\mathfrak{m}+\mathfrak{a}^n=A$. $(A/\mathfrak{m})/\left(\mathfrak{a}^n(A/\mathfrak{m})\right)=A/\left(\mathfrak{m}+\mathfrak{a}^n\right)=0$, so the $\mathfrak{a}$-adic completion of $A/\mathfrak{m}$ is 0. Hence, $A/\mathfrak{m}\rightarrow (A/\mathfrak{m})_{\hat{A}}$ isn't injective.
\end{exercise}

\begin{exercise}\label{ex10.8}
	By \Cref{ex1.5.1} and \Cref{1.6.1}, the only maximal ideal of $B$ is $\mathfrak{m}=(z_1,\cdots,z_n)$.
	
	The $\mathfrak{m}$-adic topologies on $A$, $B$ and $C$ are respectively equal to the induced topologies from the $\mathfrak{m}$-adic topology on $\mathbb{C}[z_1,\cdots,z_n]$. Apply \Cref{10.2} to $\mathbb{C}[z_1,\cdots,z_n]\subseteq A\subseteq B\subseteq C$.
	
	By \Cref{10.14}, $C$ is a flat $A$-algebra. Then use \Cref{ex10.7} and \Cref{ex3.17}.
\end{exercise}

\begin{exercise}\label{ex10.9}
	Assume that we have constructed $g_k$ and $h_k$ such that $f-g_kh_k\in\mathfrak{m}^k[x]$. Denote $w_k=f-g_kh_k$. Since there exist  $u_k$ and $v_k$ such that $1-u_kg_k-v_kh_k\in\mathfrak{m}[x]$, we have $w_k-w_ku_kg_k-w_kv_kh_k\in\mathfrak{m}^{k+1}[x]$.
	
	Applying polynomial division algorithm, we have $w_ku_k=q_kh_k+a_k$ with $\deg a_k<\deg h_k$. Since $h_k$ monic, the division process shows that  $q_k\in\mathfrak{m}^k[x]$, so $a_k\in\mathfrak{m}^k[x]$. Let $b_k=w_kv_k+q_kg_k$, then $b_k\in\mathfrak{m}^k[x]$. Note that $\deg w_k<n$, we have $\deg b_k<\deg g_k$ in the polynomial ring $\left(A/\mathfrak{m}^{k+1}\right)[x]$.
	
	Let $g_{k+1}=g_k+b_k$, $h_{k+1}=h_k+a_k$, then $f-g_{k+1}h_{k+1}\in\mathfrak{m}^{k+1}[x]$. Since $A$ complete, the coefficients of $g_k$ and $h_k$ converge, so we have $g, h\in A[x]$ such that $g\equiv g_k\mod\mathfrak{m}^k$ and $h\equiv h_k\mod\mathfrak{m}^k$. Therefore, $f-gh\in\bigcap_{k=1}^\infty\mathfrak{m}^k[x]$. By \Cref{10.17}, $\bigcap_{k=1}^\infty\mathfrak{m}^k[x]=0$.
\end{exercise}

\begin{exercise}\label{ex10.11}
	Let $A$ be the ring of germs of $C^\infty(\mathbb{R})$ functions. If $f(0)\neq0$, then $f(x)$ is always not 0 in $A$, so $\mathfrak{m}=\left\{f(x)\in A:f(0)=0\right\}$ is its only maximal ideal.
	
	$\mathfrak{m}^n=\left\{f(x)\in A:f^{(0)}(0)=\cdots=f^{(n-1)}(0)=0\right\}$, so $A/\mathfrak{m}^n\cong\mathbb{R}[x]/\left(x^n\right)$ and therefore $\hat{A}=\mathbb{R}[[x]]$. By Borel's theorem, $A\rightarrow\hat{A}$ is surjective, so $\hat{A}$ is a finitely generated $A$-module.
	
	If $A$ is Noetherian, by \Cref{10.17}, $\bigcap_{n=1}^\infty\mathfrak{m}^n=0$, but $\operatorname{e}^{-1/x^2}\in\bigcap_{n=1}^\infty\mathfrak{m}^n$.
\end{exercise}
