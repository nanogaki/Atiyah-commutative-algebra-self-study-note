\begin{exercise}\label{ex10.1}
	$B=\bigoplus_{n\geqslant0}\mathbb{Z}/p^n\mathbb{Z}$, so $A_k=\alpha^{-1}\left(p^kB\right)=\bigoplus_{n>k}\mathbb{Z}/p\mathbb{Z}$. The sequence $0\rightarrow A\rightarrow B\xrightarrow{p\cdot}B\rightarrow0$ is exact but $\hat{A}\rightarrow\hat{B}\xrightarrow{\hat{p}\cdot}\hat{B}\rightarrow0$ is not, due to the kernel of $\hat{p}\cdot$ is $\prod_{k\geqslant0}\mathbb{Z}/p\mathbb{Z}$.
\end{exercise}

\begin{exercise}
	The explicit form of the given exact sequence is
	\begin{equation*}
		0\rightarrow\bigoplus_{k>n}\mathbb{Z}/p\mathbb{Z}\rightarrow\bigoplus_{n\geqslant0}\mathbb{Z}/p\mathbb{Z}\rightarrow\bigoplus_{n=0}^k\mathbb{Z}/p\mathbb{Z}\rightarrow0.
	\end{equation*}
	Then use \Cref{ex10.1}. Analyze definitions in the proof of \Cref{10.2} to claim that $\operatorname{coker}d^B=\varprojlim^1A=0$.
\end{exercise}
